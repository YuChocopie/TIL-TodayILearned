git bash 이전 issue 모두 날리기

git branch -r | grep -Eo 'issue/.*' | xargs git push --delete downstream


git branch -av      //삭제내역 확인

다 날라가지 않은 경우 

git fetch downstream

git branch -r | grep -Eo 'issue/.*' | xargs git push --delete downstream
